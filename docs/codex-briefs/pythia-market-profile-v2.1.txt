// ============================================================
// PYTHIA — Market Profile Value Area Alerts v2.1
// Pandora's Box / Olympus Committee
// ============================================================
// v2.1 FIXES:
//   - Session detection now uses date-change method (universal
//     across ALL exchanges: NYSE, CBOE, NASDAQ, ARCA, etc.)
//   - Added fallback session detection for non-standard symbols
//   - Added session state debug info in table
//   - Fixed stale IB/VA accumulation bug on CBOE-listed ETFs
//
// Prior fixes (v2.0):
//   1. Volume distributed across bar range (not single-bin)
//   2. Prior-day VA reference + migration detection
//   3. Initial Balance tracking + IB breakout alerts
//   5. Time-based alert cooldown (not bar-based)
//   6. Poor high/low detection
//   8. Volume quality indicator in webhook payload
// ============================================================

//@version=5
indicator("Pythia Market Profile v2.1", overlay=true, max_bars_back=500)

// === INPUTS ===
sessionType      = input.string("RTH", "Session", options=["RTH", "ETH"])
numBins          = input.int(50, "Price bins", minval=20, maxval=100)
valueAreaPct     = input.float(70.0, "Value area %", minval=50, maxval=90) / 100.0
showLabels       = input.bool(true, "Show VAH/VAL/POC labels")
showPrevDay      = input.bool(true, "Show prior-day levels")
showFill         = input.bool(true, "Fill value area")
alertCooldownMin = input.int(120, "Alert cooldown (minutes)", minval=15)

// === TIME-BASED COOLDOWN ===
cooldownBars = math.ceil(alertCooldownMin / timeframe.multiplier)

// === SESSION DETECTION (v2.1 FIX — universal across exchanges) ===
// Primary: date-change detection (works for ALL symbols)
currentDate  = year * 10000 + month * 100 + dayofmonth
isNewDay     = ta.change(currentDate) != 0

// RTH hours check using hour/minute (exchange-agnostic)
utcHour   = hour(time, "America/New_York")
utcMinute = minute(time, "America/New_York")
timeInMin = utcHour * 60 + utcMinute
inRTH     = timeInMin >= 570 and timeInMin < 960  // 9:30 (570) to 16:00 (960)
inETH     = true  // ETH = all hours

inSession    = sessionType == "RTH" ? inRTH : inETH
isNewSession = isNewDay and inSession

// === INITIAL BALANCE TRACKING ===
// IB = first hour of RTH: 9:30-10:30 ET
isInIB = timeInMin >= 570 and timeInMin < 630  // 9:30 to 10:30

var float ibHigh = na
var float ibLow  = na
var bool  ibComplete = false

if isNewSession
    ibHigh     := na
    ibLow      := na
    ibComplete := false

if isInIB and inSession and not ibComplete
    ibHigh := na(ibHigh) ? high : math.max(ibHigh, high)
    ibLow  := na(ibLow)  ? low  : math.min(ibLow, low)

if not isInIB and inSession and not ibComplete and not na(ibHigh)
    ibComplete := true

// === PRIOR-DAY VA STORAGE ===
var float prevVAH = na
var float prevVAL = na
var float prevPOC = na

// === VOLUME PROFILE CALCULATION ===
var float sessionHigh = na
var float sessionLow  = na
var float[] volBins    = array.new_float(numBins, 0.0)
var float binSize      = na
var float cumSessionVol = 0.0

// Store current VA for prior-day capture
var float lastVAH = na
var float lastVAL = na
var float lastPOC = na

// Reset on new session — save prior day first
if isNewSession
    // Capture yesterday's final VA levels (use stored values, not [1] which may be stale)
    if not na(lastVAH)
        prevVAH := lastVAH
    if not na(lastVAL)
        prevVAL := lastVAL
    if not na(lastPOC)
        prevPOC := lastPOC
    sessionHigh   := high
    sessionLow    := low
    volBins       := array.new_float(numBins, 0.0)
    cumSessionVol := 0.0

// Update session range
if inSession
    sessionHigh := na(sessionHigh) ? high : math.max(sessionHigh, high)
    sessionLow  := na(sessionLow)  ? low  : math.min(sessionLow, low)

// === Distribute volume across bar's full range ===
if inSession and not na(sessionHigh) and not na(sessionLow)
    range_ = sessionHigh - sessionLow
    if range_ > 0
        binSize := range_ / numBins
        loBinIdx = math.max(0, math.floor((low - sessionLow) / binSize))
        hiBinIdx = math.min(numBins - 1, math.floor((high - sessionLow) / binSize))
        binsHit = hiBinIdx - loBinIdx + 1
        volPerBin = binsHit > 0 ? volume / binsHit : volume
        for i = loBinIdx to hiBinIdx
            curVol = array.get(volBins, i)
            array.set(volBins, i, curVol + volPerBin)
        cumSessionVol += volume

// === FIND POC ===
var int   pocBin   = 0
var float pocVol   = 0.0
var float totalVol = 0.0

if inSession and not na(binSize) and binSize > 0
    pocVol   := 0.0
    totalVol := 0.0
    for i = 0 to numBins - 1
        bv = array.get(volBins, i)
        totalVol += bv
        if bv > pocVol
            pocVol := bv
            pocBin := i

// === VALUE AREA: expand from POC ===
var int valBin = 0
var int vahBin = 0

if inSession and totalVol > 0 and not na(binSize) and binSize > 0
    vaTarget = totalVol * valueAreaPct
    vaVol    = array.get(volBins, pocBin)
    loBin    = pocBin
    hiBin    = pocBin
    loopLimit = numBins
    count = 0
    while vaVol < vaTarget and count < loopLimit
        count += 1
        loNext = loBin > 0 ? array.get(volBins, loBin - 1) : 0.0
        hiNext = hiBin < numBins - 1 ? array.get(volBins, hiBin + 1) : 0.0
        if loNext >= hiNext and loBin > 0
            loBin -= 1
            vaVol += loNext
        else if hiBin < numBins - 1
            hiBin += 1
            vaVol += hiNext
        else
            break
    valBin := loBin
    vahBin := hiBin

// === CONVERT BINS TO PRICES ===
pocPrice = not na(binSize) and binSize > 0 ? sessionLow + (pocBin + 0.5) * binSize : na
valPrice = not na(binSize) and binSize > 0 ? sessionLow + valBin * binSize : na
vahPrice = not na(binSize) and binSize > 0 ? sessionLow + (vahBin + 1) * binSize : na

// Store current VA for next session's prior-day capture
if inSession and not na(vahPrice)
    lastVAH := vahPrice
    lastVAL := valPrice
    lastPOC := pocPrice

// === Poor high/low detection ===
highBinVol = totalVol > 0 and numBins > 1 ? array.get(volBins, numBins - 1) : 0.0
lowBinVol  = totalVol > 0 ? array.get(volBins, 0) : 0.0
poorHigh = totalVol > 0 and (highBinVol / totalVol) > 0.05
poorLow  = totalVol > 0 and (lowBinVol / totalVol) > 0.05

// === Volume quality ===
avgDailyVol = ta.sma(volume, 20)
relativeVol = avgDailyVol > 0 ? cumSessionVol / (avgDailyVol * 78) : 1.0
volQuality  = relativeVol > 1.3 ? "high" : relativeVol < 0.7 ? "thin" : "normal"

// === Value area migration ===
vaMigration = "unknown"
if not na(prevVAL) and not na(prevVAH) and not na(valPrice) and not na(vahPrice)
    if valPrice > prevVAH
        vaMigration := "higher"
    else if vahPrice < prevVAL
        vaMigration := "lower"
    else
        vaMigration := "overlapping"

// === PLOT: Current session levels ===
pocPlot = plot(inSession ? pocPrice : na, "POC", color=color.new(color.yellow, 30),
     linewidth=2, style=plot.style_circles)
valPlot = plot(inSession ? valPrice : na, "VAL", color=color.new(color.green, 40),
     linewidth=1, style=plot.style_stepline)
vahPlot = plot(inSession ? vahPrice : na, "VAH", color=color.new(color.red, 40),
     linewidth=1, style=plot.style_stepline)

fill(valPlot, vahPlot, color=showFill ? color.new(color.blue, 90) : na,
     title="Value Area")

// === PLOT: Prior-day levels ===
plot(showPrevDay and inSession and not na(prevPOC) ? prevPOC : na, "Prev POC",
     color=color.new(color.yellow, 60), linewidth=1, style=plot.style_cross)
plot(showPrevDay and inSession and not na(prevVAL) ? prevVAL : na, "Prev VAL",
     color=color.new(color.green, 60), linewidth=1, style=plot.style_cross)
plot(showPrevDay and inSession and not na(prevVAH) ? prevVAH : na, "Prev VAH",
     color=color.new(color.red, 60), linewidth=1, style=plot.style_cross)

// === PLOT: Initial Balance range ===
plot(ibComplete and inSession ? ibHigh : na, "IB High",
     color=color.new(color.orange, 40), linewidth=1, style=plot.style_stepline)
plot(ibComplete and inSession ? ibLow : na, "IB Low",
     color=color.new(color.orange, 40), linewidth=1, style=plot.style_stepline)

// === LABELS ===
if showLabels and barstate.islast and inSession
    if not na(pocPrice)
        lPoc = label.new(bar_index + 2, pocPrice, "POC " + str.tostring(pocPrice, format.mintick),
             style=label.style_label_left, color=color.yellow, textcolor=color.black, size=size.small)
        label.delete(lPoc[1])
    if not na(valPrice)
        lVal = label.new(bar_index + 2, valPrice, "VAL " + str.tostring(valPrice, format.mintick),
             style=label.style_label_left, color=color.green, textcolor=color.white, size=size.small)
        label.delete(lVal[1])
    if not na(vahPrice)
        lVah = label.new(bar_index + 2, vahPrice, "VAH " + str.tostring(vahPrice, format.mintick),
             style=label.style_label_left, color=color.red, textcolor=color.white, size=size.small)
        label.delete(lVah[1])
    if showPrevDay and not na(prevPOC)
        lpPoc = label.new(bar_index + 4, prevPOC, "pPOC",
             style=label.style_label_left, color=color.new(color.yellow, 50), textcolor=color.black, size=size.tiny)
        label.delete(lpPoc[1])
    if showPrevDay and not na(prevVAL)
        lpVal = label.new(bar_index + 4, prevVAL, "pVAL",
             style=label.style_label_left, color=color.new(color.green, 50), textcolor=color.white, size=size.tiny)
        label.delete(lpVal[1])
    if showPrevDay and not na(prevVAH)
        lpVah = label.new(bar_index + 4, prevVAH, "pVAH",
             style=label.style_label_left, color=color.new(color.red, 50), textcolor=color.white, size=size.tiny)
        label.delete(lpVah[1])
    if ibComplete and not na(ibHigh)
        lIBH = label.new(bar_index + 3, ibHigh, "IB-H",
             style=label.style_label_left, color=color.orange, textcolor=color.white, size=size.tiny)
        lIBL = label.new(bar_index + 3, ibLow, "IB-L",
             style=label.style_label_left, color=color.orange, textcolor=color.white, size=size.tiny)
        label.delete(lIBH[1])
        label.delete(lIBL[1])

// === ALERT CONDITIONS ===
var int barsSinceValAlert = 999
var int barsSinceVahAlert = 999
var int barsSinceIBAlert  = 999
barsSinceValAlert += 1
barsSinceVahAlert += 1
barsSinceIBAlert  += 1

valCrossBelow = inSession and not na(valPrice) and ta.crossunder(close, valPrice)
valAlert = valCrossBelow and barsSinceValAlert > cooldownBars
if valAlert
    barsSinceValAlert := 0

vahCrossAbove = inSession and not na(vahPrice) and ta.crossover(close, vahPrice)
vahAlert = vahCrossAbove and barsSinceVahAlert > cooldownBars
if vahAlert
    barsSinceVahAlert := 0

ibBreakUp   = ibComplete and inSession and ta.crossover(close, ibHigh)
ibBreakDown = ibComplete and inSession and ta.crossunder(close, ibLow)
ibAlert = (ibBreakUp or ibBreakDown) and barsSinceIBAlert > cooldownBars
if ibAlert
    barsSinceIBAlert := 0

// === COMMON PAYLOAD FIELDS ===
comFields = '"ticker":"' + syminfo.ticker + '",' +
     '"price":' + str.tostring(close) + ',' +
     '"val":' + str.tostring(valPrice, format.mintick) + ',' +
     '"vah":' + str.tostring(vahPrice, format.mintick) + ',' +
     '"poc":' + str.tostring(pocPrice, format.mintick) + ',' +
     '"prev_val":' + str.tostring(nz(prevVAL, 0), format.mintick) + ',' +
     '"prev_vah":' + str.tostring(nz(prevVAH, 0), format.mintick) + ',' +
     '"prev_poc":' + str.tostring(nz(prevPOC, 0), format.mintick) + ',' +
     '"va_migration":"' + vaMigration + '",' +
     '"ib_high":' + str.tostring(nz(ibHigh, 0), format.mintick) + ',' +
     '"ib_low":' + str.tostring(nz(ibLow, 0), format.mintick) + ',' +
     '"poor_high":' + (poorHigh ? 'true' : 'false') + ',' +
     '"poor_low":' + (poorLow ? 'true' : 'false') + ',' +
     '"volume_quality":"' + volQuality + '"'

// === FIRE ALERTS ===
if valAlert
    valInterp = volQuality == "high" ?
         "Price at VAL with high volume - institutional buying zone" :
         "Price at VAL - watch for acceptance or rejection"
    alert('{"source":"pythia","event":"val_cross_below",' + comFields +
         ',"interpretation":"' + valInterp + '"}', alert.freq_once_per_bar)

if vahAlert
    vahInterp = volQuality == "high" ?
         "Price above VAH with high volume - potential breakout" :
         "Price above VAH in thin extension - caution"
    alert('{"source":"pythia","event":"vah_cross_above",' + comFields +
         ',"interpretation":"' + vahInterp + '"}', alert.freq_once_per_bar)

if ibAlert
    ibDir = ibBreakUp ? "above" : "below"
    ibEvent = ibBreakUp ? "ib_break_up" : "ib_break_down"
    ibInterp = ibBreakUp ?
         "IB breakout to upside - initiative buying" :
         "IB breakdown - initiative selling"
    alert('{"source":"pythia","event":"' + ibEvent + '",' + comFields +
         ',"interpretation":"' + ibInterp + '"}', alert.freq_once_per_bar)

// === VISUAL MARKERS ===
plotshape(valAlert, title="VAL Cross", location=location.belowbar,
     style=shape.triangleup, color=color.green, size=size.small, text="VAL")
plotshape(vahAlert, title="VAH Cross", location=location.abovebar,
     style=shape.triangledown, color=color.red, size=size.small, text="VAH")
plotshape(ibBreakUp and ibAlert, title="IB Break Up", location=location.abovebar,
     style=shape.diamond, color=color.orange, size=size.small, text="IB-BO")
plotshape(ibBreakDown and ibAlert, title="IB Break Down", location=location.belowbar,
     style=shape.diamond, color=color.orange, size=size.small, text="IB-BD")

plotshape(poorHigh and barstate.islast, title="Poor High", location=location.abovebar,
     style=shape.xcross, color=color.new(color.red, 30), size=size.tiny, text="PH")
plotshape(poorLow and barstate.islast, title="Poor Low", location=location.belowbar,
     style=shape.xcross, color=color.new(color.green, 30), size=size.tiny, text="PL")

// === INFO TABLE ===
var table infoTbl = table.new(position.top_right, 3, 8, bgcolor=color.new(color.black, 80))
if barstate.islast and inSession and not na(pocPrice)
    table.cell(infoTbl, 0, 0, "PYTHIA v2", text_color=color.yellow, text_size=size.small)
    table.cell(infoTbl, 1, 0, syminfo.ticker, text_color=color.white, text_size=size.small)
    table.cell(infoTbl, 2, 0, volQuality == "high" ? "HiVol" : volQuality == "thin" ? "ThinVol" : "",
         text_color=volQuality == "high" ? color.lime : color.orange, text_size=size.tiny)
    table.cell(infoTbl, 0, 1, "VAH", text_color=color.red, text_size=size.tiny)
    table.cell(infoTbl, 1, 1, str.tostring(vahPrice, format.mintick), text_color=color.white, text_size=size.tiny)
    table.cell(infoTbl, 0, 2, "POC", text_color=color.yellow, text_size=size.tiny)
    table.cell(infoTbl, 1, 2, str.tostring(pocPrice, format.mintick), text_color=color.white, text_size=size.tiny)
    table.cell(infoTbl, 0, 3, "VAL", text_color=color.green, text_size=size.tiny)
    table.cell(infoTbl, 1, 3, str.tostring(valPrice, format.mintick), text_color=color.white, text_size=size.tiny)
    table.cell(infoTbl, 0, 4, "pVAH", text_color=color.new(color.red, 40), text_size=size.tiny)
    table.cell(infoTbl, 1, 4, not na(prevVAH) ? str.tostring(prevVAH, format.mintick) : "-",
         text_color=color.gray, text_size=size.tiny)
    table.cell(infoTbl, 0, 5, "pVAL", text_color=color.new(color.green, 40), text_size=size.tiny)
    table.cell(infoTbl, 1, 5, not na(prevVAL) ? str.tostring(prevVAL, format.mintick) : "-",
         text_color=color.gray, text_size=size.tiny)
    migColor = vaMigration == "higher" ? color.lime : vaMigration == "lower" ? color.red : color.gray
    table.cell(infoTbl, 0, 6, "VA Mig", text_color=color.gray, text_size=size.tiny)
    table.cell(infoTbl, 1, 6, vaMigration, text_color=migColor, text_size=size.tiny)
    if ibComplete and not na(ibHigh)
        table.cell(infoTbl, 2, 1, "IB-H: " + str.tostring(ibHigh, format.mintick),
             text_color=color.orange, text_size=size.tiny)
        table.cell(infoTbl, 2, 2, "IB-L: " + str.tostring(ibLow, format.mintick),
             text_color=color.orange, text_size=size.tiny)
    if poorHigh
        table.cell(infoTbl, 2, 3, "POOR HIGH", text_color=color.red, text_size=size.tiny)
    if poorLow
        table.cell(infoTbl, 2, 4, "POOR LOW", text_color=color.green, text_size=size.tiny)
